  ID INT NOT NULL,
  FIRST_NAME VARCHAR(10) NOT NULL,
  LAST_NAME VARCHAR(10) NOT NULL,
  AGE INT
);

CREATE TABLE Category(
  cID INT PRIMARY KEY,
  cName VARCHAR(30) NOT NULL
);

  pID INT PRIMARY KEY,
